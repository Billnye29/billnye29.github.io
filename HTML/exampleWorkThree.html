#include <unistd.h>

#define MIN_SBRK_SIZE 8192

typedef struct BLOCK {
    int size;
    int count;
    struct BLOCK *next;
} *BLOCK;

BLOCK head = NULL;


BLOCK nalloc(size_t size) {
    BLOCK n = sbrk(size);
    n->size = size;
    n->next = NULL;
    return n;
}


void my_free(void *ptr) {
    if (ptr == NULL) {
        return;
    }
    BLOCK temp = head;
    BLOCK temp2 = head;
    BLOCK temp2Prev = NULL;
    BLOCK temp3 = (BLOCK) ((char *) ptr - 8);
    BLOCK temp4;
    if (head == NULL) {
        head = (BLOCK) ((char *) ptr - 8);
        head->next = temp;
    } else {

        temp2 = head;
        temp2Prev = NULL;

        while (1) {
            temp2Prev = temp2;
            temp2 = temp2->next;

            if (temp2 == NULL) {
                temp2Prev->next = ((char *) ptr - 8);
                temp2Prev = temp2Prev->next;
                temp2Prev->next = NULL;

                break;
            }
        }


    }

}


BLOCK newBlocks(size_t size) {
    if (size > MIN_SBRK_SIZE) {
        return (nalloc(size));
    }

    BLOCK newNode = nalloc(MIN_SBRK_SIZE);
    BLOCK temp;
    temp = newNode;
    temp = (struct BLOCK *) ((char *) newNode + size);
    temp->size = newNode->size - size;

    my_free((char *) temp + 8);
    newNode->size = size;
    return newNode;

}


void *my_malloc(size_t size) {
    int sizeFlag = 0;
    size = (size + 7 + 8) & -8;  // align on 8-byte boundary

    BLOCK newNode = head;
    BLOCK tempNode;
    BLOCK tempo = head;
    BLOCK headPre = head;
    while (1) {
        /// No blocks found so create new block
        if (tempo == NULL) {
            if (size <= MIN_SBRK_SIZE) {

                tempo = newBlocks(size);
                return (char *) tempo + 8;
            } else {
                tempo = newBlocks(size);
                return (char *) tempo + 8;
            }
        } else {
            if (tempo->size > size) {
                if (sizeFlag == 1) {
                    tempNode = (struct BLOCK *) ((char *) tempo + size);
                    tempNode->size = tempo->size - size;
                    tempNode->next = tempo->next;
                    headPre->next = tempNode;
                    tempo->size = size;
                    return (char *) tempo + 8;
                } else {
                    head = (struct BLOCK *) ((char *) tempo + size);
                    head->size = tempo->size - size;
                    head->next = tempo->next;
                    tempo->size = size;

                    return (char *) tempo + 8;
                }
            }
            if (tempo->size == size) {
                head = head->next;
                return (char *) tempo + 8;
            }
        }

        tempo = tempo->next;

        if (sizeFlag == 1) {

            headPre = headPre->next;
        }
        sizeFlag = 1;


    }
}

void *free_list_begin() {
    if (head == NULL) {
        return NULL;

    }

    return head;
}

void *free_list_next(void *node) {
    BLOCK tempo = node;
    if (tempo->next == NULL) {
        return NULL;
    }
    return tempo->next;
}

void insertionSort() {
    if (head == NULL || head->next == NULL) {
        return;
    }
    BLOCK sorted = NULL;
    BLOCK unsorted = head;
    while (unsorted != NULL) {
        BLOCK current = unsorted;
        unsorted = unsorted->next;
        if (sorted == NULL || current < sorted) {

            current->next = sorted;
            sorted = current;
        } else {
            BLOCK prev = sorted;
            while (prev->next != NULL && prev->next < current) {
                prev = prev->next;
            }
            current->next = prev->next;
            prev->next = current;
        }
    }
    head = NULL;
    head = sorted;
}

void coalesce_free_list() {
    insertionSort();
    if (head == NULL || head->next == NULL) {

        return;
    }
    BLOCK temp2prev = head;
    BLOCK temp2 = head->next;

    while (temp2 != NULL) {
        if ((char *) temp2prev + temp2prev->size == (char *) temp2) {

            temp2prev->size += temp2->size;
            temp2prev->next = temp2->next;
            temp2 = temp2prev->next;
        } else {

            temp2prev = temp2;
            temp2 = temp2->next;
        }
    }
}
